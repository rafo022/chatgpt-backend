  <div class="chat-container">

    <h2>Chat con la IA de tu Proyecto Inmobiliario</h2>

    <div id="chat-box" class="chat-box"></div>

    <input id="input-box" class="input-box" type="text" placeholder="Escribe tu mensaje..." />

    <button class="send-button" onclick="sendMessage()">Enviar</button>

  </div>


  <script>

    const chatBox = document.getElementById('chat-box');

    const inputBox = document.getElementById('input-box');


    // Función para manejar el envío del mensaje

    async function sendMessage() {

      const message = inputBox.value;

      if (!message) return;


      // Mostrar el mensaje del usuario

      chatBox.innerHTML += `<div><strong>Tú:</strong> ${message}</div>`;

      inputBox.value = ''; // Limpiar el campo de entrada


      // Enviar la solicitud al backend

      try {

        const response = await fetch('https://chat.pantalladigital.pe’/api/chat.js, {

          method: 'POST',

          headers: {

            'Content-Type': 'application/json'

          },

          body: JSON.stringify({ message })

        });


        const data = await response.json();


        if (data.reply) {

          // Mostrar la respuesta de la IA

          chatBox.innerHTML += `<div><strong>AI:</strong> ${data.reply}</div>`;

          chatBox.scrollTop = chatBox.scrollHeight; // Desplazarse hacia abajo para mostrar el nuevo mensaje

        }

      } catch (error) {

        console.error('Error al enviar el mensaje:', error);

      }

    }

  </script>
